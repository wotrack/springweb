# Using Redis 
# Set the host and port value at src/main/resources/application.properties file

# Start the Redis server , if local use 
$redis-server

# Test if Redis started successfully or not 
$ redis-cli -h localhost

# Compile and run the code
$ ./gradlew bootRun
